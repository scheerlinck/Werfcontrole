// ============================================================ // WERFCONTROLE – Google Apps Script Backend v2 // Plak deze code via Uitbreidingen → Apps Script in je Sheet // ============================================================ var SHEET_NAME = 'Werfcontrole'; var FEEDBACK_SHEET = 'Feedback'; var FOLDER_NAME = 'Werfcontrole Fotos'; var RAPPORTEN_FOLDER = 'Werfcontrole Rapporten'; // ── Hulpfuncties ────────────────────────────────────────── function getSheet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(['id','projectId','projectNaam','projectAdres','itemId','omschrijving','locatie','aannemer','done','datum','fotoUrls','aangemaaktOp']); sheet.setFrozenRows(1); } return sheet; } function getFeedbackSheet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(FEEDBACK_SHEET); if (!sheet) { sheet = ss.insertSheet(FEEDBACK_SHEET); sheet.appendRow(['rapportId','itemId','aannemer','gedaan','notitie','tijdstip']); sheet.setFrozenRows(1); } return sheet; } function getFotoFolder() { var folders = DriveApp.getFoldersByName(FOLDER_NAME); if (folders.hasNext()) return folders.next(); return DriveApp.createFolder(FOLDER_NAME); } function getRapportenFolder() { var folders = DriveApp.getFoldersByName(RAPPORTEN_FOLDER); if (folders.hasNext()) return folders.next(); return DriveApp.createFolder(RAPPORTEN_FOLDER); } function jsonResponse(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } // ── Hoofdrouter ─────────────────────────────────────────── function doGet(e) { var action = ''; try { action = e.parameter.action || ''; } catch(x) {} try { if (action === 'ping') return jsonResponse({ok: true, message: 'Werfcontrole API werkt!'}); if (action === 'getAll') return jsonResponse(getAll()); if (action === 'getRapport') return jsonResponse(getRapport(e.parameter.rapportId)); if (action === 'getFeedback') return jsonResponse(getFeedback(e.parameter.rapportId)); return jsonResponse({ok: true, message: 'Werfcontrole API actief.'}); } catch(err) { return jsonResponse({error: err.message}); } } function doPost(e) { var data = {}; try { data = JSON.parse(e.postData.contents); } catch(x) {} var action = data.action || ''; try { if (action === 'saveProject') return jsonResponse(saveProject(data)); if (action === 'saveItem') return jsonResponse(saveItem(data)); if (action === 'toggleDone') return jsonResponse(toggleDone(data)); if (action === 'deleteItem') return jsonResponse(deleteItem(data)); if (action === 'deleteProject') return jsonResponse(deleteProject(data)); if (action === 'uploadFoto') return jsonResponse(uploadFoto(data)); if (action === 'saveRapport') return jsonResponse(saveRapport(data)); if (action === 'saveFeedback') return jsonResponse(saveFeedback(data)); return jsonResponse({error: 'Onbekende actie: ' + action}); } catch(err) { return jsonResponse({error: err.message}); } } // ── Data ophalen ────────────────────────────────────────── function getAll() { var sheet = getSheet(); var rows = sheet.getDataRange().getValues(); var projects = {}; for (var i = 1; i < rows.length; i++) { var r = rows[i]; var projectId = String(r[1]); var projectNaam = String(r[2]); var projectAdres = String(r[3]); var itemId = String(r[4]); var omschrijving = String(r[5]); var locatie = String(r[6]); var aannemer = String(r[7]); var done = r[8] === true || r[8] === 'true' || r[8] === 1; var datum = String(r[9]); var fotoUrls = r[10] ? String(r[10]).split(',').filter(Boolean) : []; var aangemaaktOp = String(r[11]); if (!projectId || projectId === 'undefined') continue; if (!projects[projectId]) { projects[projectId] = { id: projectId, naam: projectNaam, adres: projectAdres, datum: aangemaaktOp || datum, items: [] }; } if (itemId && itemId !== 'undefined') { projects[projectId].items.push({ id: itemId, omschrijving: omschrijving, locatie: locatie, aannemer: aannemer, done: done, datum: datum, photos: fotoUrls }); } } var result = Object.values(projects); result.sort(function(a,b) { return b.datum > a.datum ? 1 : -1; }); return { ok: true, projects: result }; } // ── Rapport opslaan & ophalen ───────────────────────────── function saveRapport(data) { // Sla rapportdata op in Drive als JSON bestand var folder = getRapportenFolder(); var rapportId = data.rapportId; var json = JSON.stringify(data); // Verwijder oud rapport met zelfde ID als dat bestaat var files = folder.getFilesByName(rapportId + '.json'); while (files.hasNext()) files.next().setTrashed(true); var blob = Utilities.newBlob(json, 'application/json', rapportId + '.json'); var file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); return { ok: true, rapportId: rapportId }; } function getRapport(rapportId) { if (!rapportId) return { ok: false, error: 'Geen rapportId' }; var folder = getRapportenFolder(); var files = folder.getFilesByName(rapportId + '.json'); if (!files.hasNext()) return { ok: false, error: 'Rapport niet gevonden' }; var content = files.next().getBlob().getDataAsString(); var rapport = JSON.parse(content); return { ok: true, rapport: rapport }; } // ── Feedback opslaan & ophalen ──────────────────────────── function saveFeedback(data) { var sheet = getFeedbackSheet(); var rapportId = data.rapportId; var tijdstip = new Date().toLocaleString('nl-BE'); // Verwijder oude feedback voor dit rapport var rows = sheet.getDataRange().getValues(); for (var i = rows.length - 1; i >= 1; i--) { if (String(rows[i][0]) === String(rapportId)) { sheet.deleteRow(i+1); } } // Sla nieuwe feedback op per item var items = data.items || []; items.forEach(function(item) { sheet.appendRow([ rapportId, item.id, data.aannemer || '', item.gedaan ? true : false, item.notitie || '', tijdstip ]); }); return { ok: true }; } function getFeedback(rapportId) { if (!rapportId) return { ok: false, error: 'Geen rapportId' }; var sheet = getFeedbackSheet(); var rows = sheet.getDataRange().getValues(); var items = []; for (var i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(rapportId)) { items.push({ itemId: String(rows[i][1]), aannemer: String(rows[i][2]), gedaan: rows[i][3] === true || rows[i][3] === 'true', notitie: String(rows[i][4] || ''), tijdstip: String(rows[i][5] || '') }); } } return { ok: true, items: items }; } // ── Project & item beheer ───────────────────────────────── function saveProject(data) { var sheet = getSheet(); var rows = sheet.getDataRange().getValues(); var datum = new Date().toLocaleDateString('nl-BE'); for (var i = 1; i < rows.length; i++) { if (String(rows[i][1]) === String(data.id)) { sheet.getRange(i+1, 3).setValue(data.naam); sheet.getRange(i+1, 4).setValue(data.adres || ''); return { ok: true }; } } sheet.appendRow([ 'proj-' + data.id, data.id, data.naam, data.adres || '', '', '', '', '', false, datum, '', datum ]); return { ok: true }; } function saveItem(data) { var sheet = getSheet(); var rows = sheet.getDataRange().getValues(); var datum = data.datum || new Date().toLocaleDateString('nl-BE'); for (var i = 1; i < rows.length; i++) { if (String(rows[i][1]) === String(data.projectId) && !rows[i][4]) { sheet.getRange(i+1, 5).setValue(data.id); sheet.getRange(i+1, 6).setValue(data.omschrijving); sheet.getRange(i+1, 7).setValue(data.locatie || ''); sheet.getRange(i+1, 8).setValue(data.aannemer || ''); sheet.getRange(i+1, 9).setValue(false); sheet.getRange(i+1, 10).setValue(datum); sheet.getRange(i+1, 11).setValue(''); return { ok: true }; } } var projectNaam = '', projectAdres = '', projectDatum = datum; for (var j = 1; j < rows.length; j++) { if (String(rows[j][1]) === String(data.projectId)) { projectNaam = String(rows[j][2]); projectAdres = String(rows[j][3]); projectDatum = String(rows[j][11]) || datum; break; } } sheet.appendRow([ data.id, data.projectId, projectNaam, projectAdres, data.id, data.omschrijving, data.locatie || '', data.aannemer || '', false, datum, '', projectDatum ]); return { ok: true }; } function toggleDone(data) { var sheet = getSheet(); var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (String(rows[i][4]) === String(data.itemId)) { sheet.getRange(i+1, 9).setValue(data.done === true || data.done === 'true'); return { ok: true }; } } return { ok: false, error: 'Item niet gevonden' }; } function deleteItem(data) { var sheet = getSheet(); var rows = sheet.getDataRange().getValues(); for (var i = rows.length - 1; i >= 1; i--) { if (String(rows[i][4]) === String(data.itemId)) { sheet.deleteRow(i+1); return { ok: true }; } } return { ok: false, error: 'Item niet gevonden' }; } function deleteProject(data) { var sheet = getSheet(); var rows = sheet.getDataRange().getValues(); for (var i = rows.length - 1; i >= 1; i--) { if (String(rows[i][1]) === String(data.projectId)) { sheet.deleteRow(i+1); } } return { ok: true }; } function uploadFoto(data) { var folder = getFotoFolder(); var blob = Utilities.newBlob( Utilities.base64Decode(data.base64), data.mimeType || 'image/jpeg', data.filename || ('foto-' + Date.now() + '.jpg') ); var file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); var fileId = file.getId(); var viewUrl = 'https://lh3.googleusercontent.com/d/' + fileId; var sheet = getSheet(); var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (String(rows[i][4]) === String(data.itemId)) { var existing = rows[i][10] ? String(rows[i][10]) : ''; var urls = existing ? existing.split(',').filter(Boolean) : []; urls.push(viewUrl); sheet.getRange(i+1, 11).setValue(urls.join(',')); return { ok: true, url: viewUrl }; } } return { ok: true, url: viewUrl }; }
